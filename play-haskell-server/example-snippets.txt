import Automata

automaton :: AutomatonBuilder String (StackTransition String String)
automaton = do
  a <- state "a"
  b <- state "b"
  c <- state "c"

  initial a
  final c

  a >--["\\epsilon" ~~ ("\\epsilon", "\\$")]--> b

  b >--["(" ~~ ("\\epsilon", "(")]--> b
  b >--[")" ~~ ("(", "\\epsilon")]--> b

  b >--["\\epsilon" ~~ ("\\$", "\\epsilon")]--> c

  a `leftOf` b
