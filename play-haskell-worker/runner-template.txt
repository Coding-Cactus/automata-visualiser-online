-- code to append to all playground scripts
main :: IO ()
main = do
  let svgFile = "out.svg"
  let tikzFile = "out.tex"

  render config tikzFile tikz automaton
  render config svgFile svg automaton

  let sep = "==="
  putStrLn sep
  putStrLn =<< readFile svgFile
  putStrLn sep
  putStrLn =<< readFile tikzFile
